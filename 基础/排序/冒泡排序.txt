const int N = 110;
const int INF = 0x3ffffff;

int a[N];

void BubbleSort(int n)
{
    for(int i = 0; i < n-1; ++i){
        for(int j = 0; j < n-i-1; ++j){
            if(a[j] > a[j+1])
                swap(a[j],a[j+1]);
        }
    }
}

int main()
{
    srand(time(NULL));
    for(int i = 0; i < 10; ++i) a[i] = rand()%100;
    BubbleSort(10);
    for(int i = 0; i < 10; ++i)
        cout << a[i] << endl;
    return 0;
}
